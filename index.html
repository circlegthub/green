      <label for="mode">Crash mode:</label>

      <select id="mode" class="mode-select" title="Choose how the page will 'crash'">

        <option value="throw">throw (uncaught exception)</option>

        <option value="freeze">freeze (busy loop) — may hang tab</option>

        <option value="close">close (attempt to close window)</option>

      </select>

      <button id="start">Start (5s)</button>

      <div class="notice">Images must be in the same folder as this file: <code>money.jpeg</code> and <code>petsimx.jpeg</code></div>

  <script>

    (function(){

      const btn = document.getElementById('start');

      const select = document.getElementById('mode');


      // Default delay before crash (ms)

      const DELAY = 5000;


      // Helper: freeze the main thread (bad idea except for testing)

      function freezeIndefinitely() {

        // tight loop: intentionally CPU-bound

        // WARNING: this will hang the tab until the loop breaks or the tab is closed

        while (true) {

          // do small meaningless work

          Math.sqrt(Math.random() * 1e6);

        }

      }


      // Trigger an uncaught exception

      function throwCrash() {

        // throw in a microtask so it's uncaught by local try/catch

        Promise.resolve().then(() => {

          throw new Error("Intentional crash: uncaught exception");

        });

      }


      // Attempt to close the window (often blocked)

      function tryClose() {

        try {

          // approach that sometimes works if the window was opened by script:

          window.open('', '_self');

          window.close();

        } catch (e) {

          // fallback: remove page content and throw

          document.body.innerHTML = '<pre style="padding:20px">Window close blocked by browser. The page will now throw an error.</pre>';

          throwCrash();

        }

      }


      btn.addEventListener('click', () => {

        const mode = select.value;

        btn.disabled = true;

        select.disabled = true;

        btn.textContent = `Waiting ${DELAY/1000}s...`;


        setTimeout(() => {

          btn.textContent = 'Crashing now...';

          try {

            if (mode === 'throw') {

              throwCrash();

            } else if (mode === 'freeze') {

              // small delay so the UI updates before freezing

              setTimeout(freezeIndefinitely, 50);

            } else if (mode === 'close') {

              tryClose();

            } else {

              throwCrash();

            }

          } catch (err) {

            // If a synchronous throw happens, rethrow to be uncaught

            setTimeout(() => { throw err; }, 0);

          }

        }, DELAY);

      });


      // If images fail to load, show a simple message

      function hideOnError(img) {

        img.addEventListener('error', () => {

          img.style.display = 'none';

          const warn = document.createElement('div');

          warn.style.color = '#b00020';

          warn.style.marginTop = '8px';

          warn.textContent = `Failed to load ${img.alt}. Make sure "${img.src}" exists.`;

          document.querySelector('.controls').appendChild(warn);

        });

      }

      hideOnError(document.getElementById('img1'));

      hideOnError(document.getElementById('img2'));

    })();

  </script>
